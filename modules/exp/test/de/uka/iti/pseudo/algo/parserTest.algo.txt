// Testing Comments

uses "filename"

uses inline
  arbitrary text!
end

abbreviation @abbrev := true

algo Name_of_the_algorithm

input a : int
      b : int

var a0 : int
    b0 : int

requires a > 0

requires @abbrev

do
   a0 := a ;;
   b0 := b ;
   while a > 0
     inv a >= 0
     var a
   do
      a := a + 1;
      b := b - 1
   end;
   
   if a < b
   then 
     a := b
   end;

   inline
      again, arbitrary text
   end
end

refine Name_of_the_algrithm as Name_of_Implementation
requires
   a = this.a &
   b = this.b[5]
mark 1
   inv true
   var 0
ensures
   a = this.a & true
