  int a assignable
  int b assignable
  int c assignable
  int d assignable

(* a modifies analysis of [0;P] ought to yield {c,a} *)

  S: goto A, S2
  S2: goto B, S3
  S3: goto C, D
  
  A: a := 0
     goto S
  B: b := 0
     goto B
     goto S
  C: c := 0
     goto A, B
  D: d := 0
     end true
     goto S
